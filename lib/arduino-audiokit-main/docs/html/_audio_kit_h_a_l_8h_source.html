85  /// @brief driver which selects the codec implementation: AUDIO_CODEC_ES8388_DEFAULT_HANDLE, AUDIO_CODEC_AC101_CODEC_HANDLE, AUDIO_CODEC_ES8311_DEFAULT_HANDLE, AUDIO_CODEC_ES7243_DEFAULT_HANDLE

95 
96  audio_hal_adc_input_t adc_input
97  = AUDIOKIT_DEFAULT_INPUT; /*!< set adc channel with audio_hal_adc_input_t*/
98  audio_hal_dac_output_t dac_output = AUDIOKIT_DEFAULT_OUTPUT; /*!< set dac channel */
99  audio_hal_codec_mode_t codec_mode; /*!< select codec mode: adc, dac or both */
100  audio_hal_iface_mode_t master_slave_mode
101  = AUDIOKIT_DEFAULT_MASTER_SLAVE; /*!< audio codec chip mode */
102  audio_hal_iface_format_t fmt = AUDIOKIT_DEFAULT_I2S_FMT; /*!< I2S interface format */
103  audio_hal_iface_samples_t sample_rate
104  = AUDIOKIT_DEFAULT_RATE; /*!< I2S interface samples per second */
105  audio_hal_iface_bits_t bits_per_sample
106  = AUDIOKIT_DEFAULT_BITSIZE; /*!< i2s interface number of bits per sample */
107 
108  /// Returns true if the CODEC is the master
109  bool isMaster() { return master_slave_mode == AUDIO_HAL_MODE_MASTER; }
110 
111  /// provides the bits per sample
112  int bitsPerSample()
113  {
114  switch (bits_per_sample) {
115  case AUDIO_HAL_BIT_LENGTH_16BITS:
116  return 16;
117  case AUDIO_HAL_BIT_LENGTH_24BITS:
118  return 24;
119  case AUDIO_HAL_BIT_LENGTH_32BITS:
120  return 32;
121  }
122  KIT_LOGE("bits_per_sample not supported: %d", bits_per_sample);
123  return 0;
124  }
125 
126  /// Provides the sample rate in samples per second
127  uint32_t sampleRate()
128  {
129  switch (sample_rate) {
130  case AUDIO_HAL_08K_SAMPLES: /*!< set to 8k samples per second */
131  return 8000;
132  case AUDIO_HAL_11K_SAMPLES: /*!< set to 11.025k samples per second */
133  return 11025;
134  case AUDIO_HAL_16K_SAMPLES: /*!< set to 16k samples in per second */
135  return 16000;
136  case AUDIO_HAL_22K_SAMPLES: /*!< set to 22.050k samples per second */
137  return 22050;
138  case AUDIO_HAL_24K_SAMPLES: /*!< set to 24k samples in per second */
139  return 24000;
140  case AUDIO_HAL_32K_SAMPLES: /*!< set to 32k samples in per second */
141  return 32000;
142  case AUDIO_HAL_44K_SAMPLES: /*!< set to 44.1k samples per second */
143  return 44100;
144  case AUDIO_HAL_48K_SAMPLES: /*!< set to 48k samples per second */
145  return 48000;
146  }
147  KIT_LOGE("sample rate not supported: %d", sample_rate);
148  return 0;
149  }

181 protected:
182  i2s_comm_format_t i2sFormat()
183  {
184  i2s_comm_format_t its_com_fmt = (i2s_comm_format_t)I2S_COMM_FORMAT_STAND_I2S;
185  if (fmt == AUDIO_HAL_I2S_LEFT) {
186  its_com_fmt = (i2s_comm_format_t)I2S_COMM_FORMAT_STAND_MSB;
187  } else if (fmt == AUDIO_HAL_I2S_RIGHT) {
188  its_com_fmt = (i2s_comm_format_t)I2S_COMM_FORMAT_STAND_MSB;
189  } else if (fmt == AUDIO_HAL_I2S_DSP) {
190  its_com_fmt = (i2s_comm_format_t)I2S_COMM_FORMAT_STAND_PCM_SHORT;
191  }
192  return its_com_fmt;
193  }
194 
195  i2s_mode_t i2sMode()
196  {
197  int mode = isMaster() ? I2S_MODE_SLAVE : I2S_MODE_MASTER;
198  // using ESP32 dac/adc
199  if (fmt == AUDIO_HAL_I2S_DSP) {
200 #if defined(CONFIG_IDF_TARGET_ESP32S3) || defined(CONFIG_IDF_TARGET_ESP32S2) \
201  || defined(CONFIG_IDF_TARGET_ESP32C3)
202  KIT_LOGE("AUDIO_HAL_I2S_DSP not supported");
203 #else
204  if (codec_mode == AUDIO_HAL_CODEC_MODE_DECODE) {
205  mode = mode | I2S_MODE_TX | I2S_MODE_DAC_BUILT_IN;
206  } else if (codec_mode == AUDIO_HAL_CODEC_MODE_ENCODE) {
207  mode = mode | I2S_MODE_RX | I2S_MODE_ADC_BUILT_IN;
208  } else if (codec_mode == AUDIO_HAL_CODEC_MODE_BOTH) {
209  mode = mode | I2S_MODE_RX | I2S_MODE_TX | I2S_MODE_ADC_BUILT_IN
210  | I2S_MODE_DAC_BUILT_IN;
211  }
212 #endif
213  } else {
214  // I2S
215  if (codec_mode == AUDIO_HAL_CODEC_MODE_DECODE) {
216  mode = mode | I2S_MODE_TX;
217  } else if (codec_mode == AUDIO_HAL_CODEC_MODE_ENCODE) {
218  mode = mode | I2S_MODE_RX;
219  } else if (codec_mode == AUDIO_HAL_CODEC_MODE_BOTH) {
220  mode = mode | I2S_MODE_RX | I2S_MODE_TX;
221  }
222  }
223  return (i2s_mode_t)mode;
224  }
225 
226 #endif

267  /// Starts the codec
268  bool begin(AudioKitConfig config)
269  {
270  KIT_LOGI(LOG_METHOD);
271  cfg = config;
272 
273  KIT_LOGI("Selected board: %d", AUDIOKIT_BOARD);
274  log_i2c();
275 
276  // setup driver pins
277  driver.setup(cfg.pins);
278  setupDriverPins();
279 
280  // if already active we stop first
281  if (is_active) {
282  end();
283  }
284 
285 #if AUDIOKIT_SETUP_SD
286  // release SPI for SD card if it is not necessary
287  if (AUDIOKIT_SETUP_SD && !cfg.sd_active) {
288  p_spi->end();
289  }
290  setupSPI();
291 #endif
292 
293  // setup headphone if necessary
294  setupHeadphoneDetection();
295 
296  audio_hal_conf.adc_input = cfg.adc_input;
297  audio_hal_conf.dac_output = cfg.dac_output;
298  audio_hal_conf.codec_mode = cfg.codec_mode;
299  audio_hal_conf.i2s_iface.mode = cfg.master_slave_mode;
300  audio_hal_conf.i2s_iface.fmt = cfg.fmt;
301  audio_hal_conf.i2s_iface.samples = cfg.sample_rate;
302  audio_hal_conf.i2s_iface.bits = cfg.bits_per_sample;
303 
304  // init HAL
305  if (hal_handle == 0) {
306  hal_handle = audio_hal_init(&audio_hal_conf, &cfg.driver);
307  if (hal_handle == 0) {
308  KIT_LOGE("audio_hal_init");
309  return false;
310  }
311  }
312 
313  if (cfg.i2s_active) {
314  setupI2S(cfg);
315  }
316 
317  // call start
318  if (!setActive(true)) {
319  KIT_LOGE("setActive");
320  return false;
321  }
322 
323  // activate speaker when output not earphones only
324  setSpeakerActive(cfg.dac_output != AUDIO_HAL_DAC_OUTPUT_LINE1);
325 
326  is_active = true;
327  return true;
328  }
329 
330  /// Stops the CODEC
331  bool end()
332  {
333  KIT_LOGI(LOG_METHOD);
334 
335 #if AUDIOKIT_ESP32_I2S && defined(ESP32)
336  if (cfg.i2s_active) {
337  // uninstall i2s driver
338  i2s_driver_uninstall(cfg.i2s_num);
339  }
340 #endif
341  // stop codec driver
342  audio_hal_ctrl_codec(hal_handle, cfg.codec_mode, AUDIO_HAL_CTRL_STOP);
343  // deinit
344  audio_hal_deinit(hal_handle);
345  // this will save power
346  setSpeakerActive(false);
347 
348  is_active = false;
349  hal_handle = 0;
350  return true;
351  }
